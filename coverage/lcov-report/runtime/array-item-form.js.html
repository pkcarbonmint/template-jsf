/**
 * ArrayItemForm Module
 * Manages the form for adding/editing array items in a modal dialog
 */
 
const Mustache = require('mustache');
const fs = require('fs');
const path = require('path');
 
class ArrayItemForm {
  constructor() {
    // Load the array item form template
    this.template = fs.readFileSync(path.resolve(__dirname, '../templates/array-item-form.mustache'), 'utf8');
    
    // Current state
    this.activeTable = null;
    this.activeRow = null;
    this.currentItemSchema = null;
    this.isEditMode = false;
    this.editIndex = -1;
  }
  
  /**
   * Open the form modal for adding/editing an array item
   * @param {Object} schema - The schema for the array items
   * @param {HTMLElement} table - The array table element
   * @param {HTMLElement} row - The row being edited (if in edit mode)
   * @param {Object} initialData - Initial data for the form (if editing)
   */
  openForm(schema, table, row = null, initialData = null) {
    if (!schema || !table) {
      console.error('Schema and table are required to open array item form');
      return;
    }
    
    // Store state
    this.activeTable = table;
    this.activeRow = row;
    this.currentItemSchema = schema.items || {};
    this.isEditMode = !!row;
    
    if (this.isEditMode && row) {
      // Find the row index
      const tbody = table.querySelector('tbody');
      const rows = Array.from(tbody.querySelectorAll('tr:not(.array-row-template)'));
      this.editIndex = rows.indexOf(row);
    } else {
      this.editIndex = -1;
    }
    
    // Generate unique IDs
    const timestamp = Date.now();
    const formId = `array-item-form-${timestamp}`;
    const modalId = `array-item-modal-${timestamp}`;
    
    // Get modal title based on mode
    const itemLabel = schema.layout?.itemLabel || 'Item';
    const modalTitle = this.isEditMode ? `Edit ${itemLabel}` : `Add ${itemLabel}`;
    
    // Prepare form fields
    const fields = [];
    
    if (this.currentItemSchema.type === 'object' && this.currentItemSchema.properties) {
      const properties = this.currentItemSchema.properties;
      const required = this.currentItemSchema.required || [];
      
      Object.entries(properties).forEach(([fieldName, property]) => {
        // Skip fields that should be hidden
        if (property.ui && property.ui.hidden === true) {
          return;
        }
        
        // Create field data
        const fieldId = `${formId}-${fieldName}`;
        const field = {
          fieldId,
          fieldName,
          field: fieldName,
          label: property.title || this.formatLabel(fieldName),
          description: property.description,
          required: required.includes(fieldName),
          
          // Type flags
          isBoolean: property.type === 'boolean',
          isNumber: property.type === 'number',
          isInteger: property.type === 'integer',
          isString: property.type === 'string',
          isEnum: property.enum != null,
          
          // String format flags
          isEmail: property.type === 'string' && property.format === 'email',
          isDate: property.type === 'string' && property.format === 'date',
          isUrl: property.type === 'string' && property.format === 'uri',
          isTextarea: property.type === 'string' && (property.format === 'textarea' || property.maxLength > 100),
          
          // Constraints
          min: property.minimum,
          max: property.maximum,
          step: property.multipleOf,
          pattern: property.pattern
        };
        
        // Handle initial values if editing
        if (initialData && initialData[fieldName] !== undefined) {
          field.defaultValue = initialData[fieldName];
        } else if (property.default !== undefined) {
          field.defaultValue = property.default;
        }
        
        // Handle enum options
        if (field.isEnum && property.enum) {
          field.options = property.enum.map(value => {
            // Check if enum has a mapping
            const label = property.enumNames ? 
              (property.enumNames[property.enum.indexOf(value)] || value) : 
              value;
            
            return {
              value,
              label,
              selected: field.defaultValue === value
            };
          });
        }
        
        fields.push(field);
      });
    } else if (this.currentItemSchema.type === 'string') {
      // Simple string item
      const fieldId = `${formId}-value`;
      
      fields.push({
        fieldId,
        fieldName: 'value',
        field: 'value',
        label: 'Value',
        isString: true,
        required: true,
        defaultValue: initialData || ''
      });
    }
    
    // Render the form
    const html = Mustache.render(this.template, {
      modalId,
      formId,
      modalTitle,
      fields
    });
    
    // Add the form to the DOM
    this.appendFormToDOM(html);
    
    // Set up event listeners for the form
    this.setupFormEventListeners(modalId, formId);
  }
  
  /**
   * Append the form HTML to the DOM
   * @param {string} html - The form HTML
   */
  appendFormToDOM(html) {
    // Create a container div
    const container = document.createElement('div');
    container.innerHTML = html;
    
    // Append to body
    document.body.appendChild(container.firstElementChild);
  }
  
  /**
   * Set up event listeners for the form
   * @param {string} modalId - The modal ID
   * @param {string} formId - The form ID
   */
  setupFormEventListeners(modalId, formId) {
    const modal = document.getElementById(modalId);
    const form = document.getElementById(formId);
    
    if (!modal || !form) return;
    
    // Close button
    const closeBtn = modal.querySelector('.modal-close');
    const cancelBtn = modal.querySelector('.btn-cancel');
    const saveBtn = modal.querySelector('.btn-save');
    const overlay = modal.querySelector('.modal-overlay');
    
    if (closeBtn) {
      closeBtn.addEventListener('click', () => this.closeForm(modal));
    }
    
    if (cancelBtn) {
      cancelBtn.addEventListener('click', () => this.closeForm(modal));
    }
    
    if (overlay) {
      overlay.addEventListener('click', (e) => {
        if (e.target === overlay) {
          this.closeForm(modal);
        }
      });
    }
    
    if (saveBtn) {
      saveBtn.addEventListener('click', () => this.saveForm(form, modal));
    }
    
    // Allow Enter key to submit
    form.addEventListener('keydown', (e) => {
      if (e.key === 'Enter' && !e.shiftKey) {
        e.preventDefault();
        this.saveForm(form, modal);
      }
    });
    
    // Show the modal
    modal.classList.remove('hidden');
    
    // Focus first input
    const firstInput = form.querySelector('input, select, textarea');
    if (firstInput) {
      firstInput.focus();
    }
  }
  
  /**
   * Close the form modal
   * @param {HTMLElement} modal - The modal element
   */
  closeForm(modal) {
    if (!modal) return;
    
    // Hide and remove the modal after animation
    modal.classList.add('hidden');
    
    // Use setTimeout to allow for animation
    setTimeout(() => {
      modal.remove();
    }, 300);
    
    // Clear state
    this.activeTable = null;
    this.activeRow = null;
    this.currentItemSchema = null;
    this.isEditMode = false;
    this.editIndex = -1;
  }
  
  /**
   * Save the form data
   * @param {HTMLElement} form - The form element
   * @param {HTMLElement} modal - The modal element
   */
  saveForm(form, modal) {
    if (!form || !this.activeTable || !this.currentItemSchema) return;
    
    // Validate the form
    if (!this.validateForm(form)) {
      return;
    }
    
    // Collect form data
    const formData = this.collectFormData(form);
    
    // Update or add row
    if (this.isEditMode && this.activeRow) {
      this.updateTableRow(formData);
    } else {
      this.addTableRow(formData);
    }
    
    // Close the form
    this.closeForm(modal);
  }
  
  /**
   * Validate the form
   * @param {HTMLElement} form - The form element
   * @returns {boolean} Whether the form is valid
   */
  validateForm(form) {
    if (!form || !this.currentItemSchema) return false;
    
    let isValid = true;
    const required = this.currentItemSchema.required || [];
    
    // Clear previous errors
    const errorElements = form.querySelectorAll('.field-error');
    errorElements.forEach(el => {
      el.textContent = '';
      el.classList.add('hidden');
    });
    
    // Check each field
    if (this.currentItemSchema.type === 'object' && this.currentItemSchema.properties) {
      const properties = this.currentItemSchema.properties;
      
      Object.entries(properties).forEach(([fieldName, property]) => {
        const input = form.querySelector(`[data-field="${fieldName}"]`);
        if (!input) return;
        
        const errorEl = input.closest('.form-field').querySelector('.field-error');
        if (!errorEl) return;
        
        // Check required
        if (required.includes(fieldName)) {
          let isEmpty = false;
          
          if (input.type === 'checkbox') {
            // For checkboxes, we consider them "filled" by default
            isEmpty = false;
          } else {
            isEmpty = !input.value.trim();
          }
          
          if (isEmpty) {
            errorEl.textContent = 'This field is required';
            errorEl.classList.remove('hidden');
            isValid = false;
          }
        }
        
        // Check pattern if not empty
        if (input.value && property.pattern && input.getAttribute('pattern')) {
          const regex = new RegExp(property.pattern);
          if (!regex.test(input.value)) {
            errorEl.textContent = 'Invalid format';
            errorEl.classList.remove('hidden');
            isValid = false;
          }
        }
        
        // Check numeric constraints
        if ((property.type === 'number' || property.type === 'integer') && input.value) {
          const value = parseFloat(input.value);
          
          if (property.minimum !== undefined && value < property.minimum) {
            errorEl.textContent = `Minimum value is ${property.minimum}`;
            errorEl.classList.remove('hidden');
            isValid = false;
          }
          
          if (property.maximum !== undefined && value > property.maximum) {
            errorEl.textContent = `Maximum value is ${property.maximum}`;
            errorEl.classList.remove('hidden');
            isValid = false;
          }
        }
      });
    } else if (this.currentItemSchema.type === 'string') {
      // Simple string item
      const input = form.querySelector('[data-field="value"]');
      if (input) {
        const errorEl = input.closest('.form-field').querySelector('.field-error');
        if (errorEl && !input.value.trim()) {
          errorEl.textContent = 'This field is required';
          errorEl.classList.remove('hidden');
          isValid = false;
        }
      }
    }
    
    return isValid;
  }
  
  /**
   * Collect form data
   * @param {HTMLElement} form - The form element
   * @returns {Object|string} The collected data
   */
  collectFormData(form) {
    if (!form || !this.currentItemSchema) return null;
    
    if (this.currentItemSchema.type === 'object') {
      const data = {};
      
      // Collect data from each field
      Object.entries(this.currentItemSchema.properties || {}).forEach(([fieldName, property]) => {
        const input = form.querySelector(`[data-field="${fieldName}"]`);
        if (!input) return;
        
        let value;
        
        if (input.type === 'checkbox') {
          value = input.checked;
        } else if (property.type === 'number' || property.type === 'integer') {
          value = input.value !== '' ? Number(input.value) : null;
        } else {
          value = input.value;
        }
        
        data[fieldName] = value;
      });
      
      return data;
    } else if (this.currentItemSchema.type === 'string') {
      // Simple string item
      const input = form.querySelector('[data-field="value"]');
      return input ? input.value : '';
    }
    
    return null;
  }
  
  /**
   * Update an existing table row with form data
   * @param {Object|string} data - The form data
   */
  updateTableRow(data) {
    if (!this.activeTable || !this.activeRow || !data) return;
    
    if (typeof data === 'object') {
      // Update object item
      Object.entries(data).forEach(([key, value]) => {
        const cell = this.activeRow.querySelector(`td[data-field="${key}"]`);
        if (!cell) return;
        
        const input = cell.querySelector('input, select, textarea');
        if (!input) return;
        
        if (input.type === 'checkbox') {
          input.checked = Boolean(value);
        } else {
          input.value = value !== null && value !== undefined ? value : '';
        }
      });
    } else {
      // Update simple string item
      const valueCell = this.activeRow.querySelector('td[data-field="value"]');
      if (valueCell) {
        const input = valueCell.querySelector('input');
        if (input) {
          input.value = data;
        }
      }
    }
    
    // Trigger array change event
    this.triggerArrayChangeEvent();
  }
  
  /**
   * Add a new table row with form data
   * @param {Object|string} data - The form data
   */
  addTableRow(data) {
    if (!this.activeTable || !data) return;
    
    // Call the array table handlers to add a new row
    if (window.ArrayTableHandlers) {
      const newRow = window.ArrayTableHandlers.addArrayRow(this.activeTable);
      
      if (newRow) {
        if (typeof data === 'object') {
          // Set object item data
          Object.entries(data).forEach(([key, value]) => {
            const cell = newRow.querySelector(`td[data-field="${key}"]`);
            if (!cell) return;
            
            const input = cell.querySelector('input, select, textarea');
            if (!input) return;
            
            if (input.type === 'checkbox') {
              input.checked = Boolean(value);
            } else {
              input.value = value !== null && value !== undefined ? value : '';
            }
          });
        } else {
          // Set simple string item data
          const valueCell = newRow.querySelector('td[data-field="value"]');
          if (valueCell) {
            const input = valueCell.querySelector('input');
            if (input) {
              input.value = data;
            }
          }
        }
      }
    }
  }
  
  /**
   * Trigger array change event
   */
  triggerArrayChangeEvent() {
    if (!this.activeTable) return;
    
    // Create a custom event
    const event = new CustomEvent('array:change', {
      bubbles: true,
      cancelable: true,
      detail: {
        arrayId: this.activeTable.id,
        table: this.activeTable
      }
    });
    
    this.activeTable.dispatchEvent(event);
  }
  
  /**
   * Format a field name as a human-readable label
   * @param {string} fieldName - The field name
   * @returns {string} The formatted label
   */
  formatLabel(fieldName) {
    if (!fieldName) return '';
    
    // Convert camelCase or snake_case to words with spaces
    return fieldName
      .replace(/([a-z])([A-Z])/g, '$1 $2') // camelCase to spaces
      .replace(/_/g, ' ') // snake_case to spaces
      .replace(/\b\w/g, l => l.toUpperCase()); // Capitalize first letter of each word
  }
}
